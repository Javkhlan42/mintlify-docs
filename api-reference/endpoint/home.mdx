## Error Handling

<Tabs>
  <Tab title="Хуудас ачааллахгүй">
    ```javascript
    async function loadHomePageWithRetry(maxRetries = 3) {
      for (let i = 0; i < maxRetries; i++) {
        try {
          const response = await fetch('https://sharnom.systems/');
          if (response.ok) {
            return await response.text();
          }
        } catch (error) {
          console.error(`Оролдлого ${i + 1} амжилтгүй:`, error);
          
          if (i === maxRetries - 1) {
            throw new Error('Хуудас ачааллаж чадсангүй');
          }
          
          // 1 секунд хүлээх
          await new Promise(resolve => setTimeout(resolve, 1000));
        }
      }
    }
    ```
  </Tab>
  
  <Tab title="Timeout handling">
    ```javascript
    async function loadWithTimeout(timeout = 5000) {
      const controller = new AbortController();
      const id = setTimeout(() => controller.abort(), timeout);
      
      try {
        const response = await fetch('https://sharnom.systems/', {
          signal: controller.signal
        });
        clearTimeout(id);
        return await response.text();
      } catch (error) {
        if (error.name === 'AbortError') {
          throw new Error('Request timeout');
        }
        throw error;
      }
    }
    ```
  </Tab>
</Tabs>
